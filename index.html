const compressImage = (file) => {

  return new Promise((resolve) => {

    const reader = new FileReader();

    reader.readAsDataURL(file);

    reader.onload = (event) => {

      const img = new Image();

      img.src = event.target.result;

      img.onload = () => {

        const canvas = document.createElement('canvas');

        const MAX_WIDTH = 600;

        const MAX_HEIGHT = 600;

        let width = img.width;

        let height = img.height;


        if (width > height && width > MAX_WIDTH) { height *= MAX_WIDTH / width; width = MAX_WIDTH; } 

        else if (height > MAX_HEIGHT) { width *= MAX_HEIGHT / height; height = MAX_HEIGHT; }

        

        canvas.width = width; canvas.height = height;

        const ctx = canvas.getContext('2d');

        ctx.drawImage(img, 0, 0, width, height);

        resolve(canvas.toDataURL('image/jpeg', 0.5)); 

      };

    };

  });

};

function KasirView({ products, transactions, onSaveTransaction, formatRupiah, searchQuery, settings, editingTrx, clearEditingTrx }) {

  const [cart, setCart] = useState([]);

  const [customerName, setCustomerName] = useState(''); const [customerPhone, setCustomerPhone] = useState(''); const [pickupDate, setPickupDate] = useState(new Date(Date.now() + 86400000 * 2).toISOString().split('T')[0]);

  const [trxId, setTrxId] = useState(null); const [receiptData, setReceiptData] = useState(null);


  useEffect(() => {

    if (editingTrx) { setCart(editingTrx.items); setCustomerName(editingTrx.customer); setCustomerPhone(editingTrx.customerPhone || ''); setPickupDate(editingTrx.pickupDate || ''); setTrxId(editingTrx.id); }

  }, [editingTrx]);


  const cancelEdit = () => { setCart([]); setCustomerName(''); setCustomerPhone(''); setPickupDate(new Date(Date.now() + 86400000 * 2).toISOString().split('T')[0]); setTrxId(null); clearEditingTrx(); };

  const filteredProducts = products.filter(p => p.name.toLowerCase().includes(searchQuery.toLowerCase()));

  const addToCart = (product) => { setCart(cart.find(i => i.id === product.id) ? cart.map(i => i.id === product.id ? { ...i, qty: (parseFloat(i.qty) || 0) + 1 } : i) : [...cart, { ...product, qty: 1 }]); };

  const updateQty = (id, delta) => { setCart(cart.map(i => { if (i.id === id) { const n = (parseFloat(i.qty) || 0) + delta; return n > 0 ? { ...i, qty: n } : null; } return i; }).filter(Boolean)); };

  const handleQtyChange = (id, val) => setCart(cart.map(i => i.id === id ? { ...i, qty: val } : i));

  const removeCartItem = (id) => setCart(cart.filter(i => i.id !== id));

  const totalCart = cart.reduce((sum, item) => sum + (item.price * (parseFloat(item.qty) || 0)), 0);


  const handleCheckout = (status) => {

    if (cart.length === 0) return;

    const cleanCart = cart.map(i => ({...i, qty: parseFloat(i.qty) || 0})).filter(i => i.qty > 0);

    const newTrx = { id: trxId || `TRX-${Date.now().toString().slice(-6)}`, customer: customerName || 'Pelanggan Baru', customerPhone: customerPhone, items: cleanCart, total: totalCart, status: status, date: trxId && editingTrx ? editingTrx.date : new Date().toISOString(), pickupDate: pickupDate };

    onSaveTransaction(newTrx); setReceiptData(newTrx); cancelEdit();

  };


  return (

    <div className="flex flex-row gap-3 sm:gap-6 animate-in fade-in h-[calc(100vh-140px)]">

      <div className="flex-1 overflow-y-auto pr-1 pb-4">

        <h2 className="text-lg font-bold text-gray-800 mb-4">Pilih Layanan</h2>

        <div className="grid grid-cols-2 md:grid-cols-3 xl:grid-cols-4 gap-2 sm:gap-3">

          {filteredProducts.map(p => (

            <button key={p.id} onClick={() => addToCart(p)} className="bg-white rounded-xl border p-2 flex flex-col items-center text-center hover:border-blue-500 group">

              <div className="w-12 h-12 bg-blue-50 rounded-full flex items-center justify-center mb-2 overflow-hidden">{p.image ? <img src={p.image} className="w-full h-full object-cover" /> : <Package size={24} className="text-blue-500" />}</div>

              <h3 className="font-medium text-xs mb-1 line-clamp-2">{p.name}</h3><p className="text-blue-600 font-bold text-xs mb-2">{formatRupiah(p.price)}</p>

              <div className="mt-auto w-full py-1 bg-blue-50 text-blue-600 text-[10px] font-bold rounded">+ Tambah</div>

            </button>

          ))}

        </div>

      </div>


      <div className="w-[280px] sm:w-[340px] bg-white rounded-2xl shadow-lg border flex flex-col shrink-0 h-full">

        <div className="p-3 border-b bg-gray-50 rounded-t-2xl space-y-2">

          <div className="flex justify-between items-center"><h2 className="font-bold text-gray-800">Pesanan {trxId ? 'Diedit' : 'Saat Ini'}</h2>{trxId && <button onClick={cancelEdit} className="text-red-500 text-xs font-bold"><XCircle size={14} className="inline"/> Batal</button>}</div>

          <input type="text" placeholder="Nama Pelanggan" value={customerName} onChange={e=>setCustomerName(e.target.value)} className="w-full px-2 py-1.5 text-sm border rounded focus:ring-1 outline-none" />

          <input type="text" placeholder="No. WA (0812...)" value={customerPhone} onChange={e=>setCustomerPhone(e.target.value)} className="w-full px-2 py-1.5 text-sm border rounded focus:ring-1 outline-none" />

          <div className="flex items-center gap-2 bg-white px-2 py-1.5 border rounded text-sm"><CalendarDays size={14} className="text-gray-400"/><span className="text-gray-500 text-xs flex-1">Ambil:</span><input type="date" value={pickupDate} onChange={e=>setPickupDate(e.target.value)} className="outline-none text-right font-medium cursor-pointer bg-transparent"/></div>

        </div>

        <div className="flex-1 overflow-y-auto p-3 space-y-2">

          {cart.length===0 ? <div className="h-full flex flex-col items-center justify-center text-gray-400 text-sm"><ShoppingCart size={32} className="mb-2 opacity-50"/>Keranjang Kosong</div> : 

           cart.map(i => (

            <div key={i.id} className="bg-gray-50 p-2 rounded border"><div className="flex justify-between items-start mb-2"><div className="flex-1"><h4 className="text-xs font-bold leading-tight">{i.name}</h4><p className="text-[10px] text-gray-500">{formatRupiah(i.price)}/{i.unit}</p></div><button onClick={()=>removeCartItem(i.id)} className="text-red-400 p-1"><Trash2 size={12}/></button></div><div className="flex justify-between items-center"><div className="flex bg-white border rounded"><button onClick={()=>updateQty(i.id, -1)} className="px-2 bg-gray-100">-</button><input type="number" value={i.qty} onChange={e=>handleQtyChange(i.id, e.target.value)} className="w-10 text-center text-xs font-bold border-x py-0.5 outline-none"/><button onClick={()=>updateQty(i.id, 1)} className="px-2 bg-gray-100">+</button></div><p className="text-xs font-bold">{formatRupiah(i.price*(parseFloat(i.qty)||0))}</p></div></div>

          ))}

        </div>

        <div className="p-3 border-t bg-white rounded-b-2xl">

          <div className="flex justify-between items-center mb-3"><span className="text-sm font-medium">Total Tagihan</span><span className="text-xl font-bold">{formatRupiah(totalCart)}</span></div>

          <div className="grid grid-cols-2 gap-2"><button onClick={()=>handleCheckout('Belum Lunas')} disabled={cart.length===0} className="py-2 bg-orange-100 text-orange-700 font-semibold rounded-xl text-xs flex flex-col items-center"><Clock size={16}/>Simpan (Tunda)</button><button onClick={()=>handleCheckout('Lunas')} disabled={cart.length===0} className="py-2 bg-blue-600 text-white font-semibold rounded-xl text-xs flex flex-col items-center"><CheckCircle2 size={16}/>Bayar Lunas</button></div>

        </div>

      </div>

      {receiptData && <ReceiptModal data={receiptData} settings={settings} onClose={()=>setReceiptData(null)} formatRupiah={formatRupiah} />}

    </div>

  );

}


function LaporanView({ transactions, onDeleteTransaction, formatRupiah, searchQuery, onLoadToKasir, settings }) {

  const [startDate, setStartDate] = useState(''); const [endDate, setEndDate] = useState('');

  const filteredTrx = transactions.filter(t => {

    if (!t.id.toLowerCase().includes(searchQuery.toLowerCase()) && !t.customer.toLowerCase().includes(searchQuery.toLowerCase())) return false;

    const d = new Date(t.date); if (startDate && d < new Date(startDate)) return false; if (endDate && d > new Date(endDate+"T23:59:59")) return false;

    return true;

  }).sort((a,b) => new Date(b.date) - new Date(a.date));

  const totalOmset = filteredTrx.filter(t => t.status === 'Lunas').reduce((s, t) => s + t.total, 0);


  return (

    <div className="space-y-4 animate-in fade-in">

      <div className="flex justify-between items-center print:hidden"><h2 className="text-xl font-bold text-gray-800">Laporan</h2><button onClick={()=>window.print()} className="bg-green-600 text-white px-3 py-1.5 rounded flex items-center gap-1 text-sm"><Download size={14}/> Cetak PDF</button></div>

      <div className="bg-white p-3 rounded-lg shadow-sm border flex gap-3 print:hidden text-sm"><input type="date" value={startDate} onChange={e=>setStartDate(e.target.value)} className="border rounded px-2 py-1 outline-none"/><input type="date" value={endDate} onChange={e=>setEndDate(e.target.value)} className="border rounded px-2 py-1 outline-none"/>{(startDate||endDate)&&<button onClick={()=>{setStartDate('');setEndDate('');}} className="text-xs text-red-500 hover:underline">Reset</button>}</div>

      <div className="hidden print:block text-center border-b-2 pb-2 mb-4"><h1 className="text-xl font-bold">{settings.storeName}</h1><p className="text-xs text-gray-600">Laporan Transaksi</p></div>

      <div className="bg-white rounded-lg border overflow-x-auto"><table className="w-full text-left text-sm print:text-xs"><thead><tr className="bg-gray-50 border-b"><th className="p-3">Pelanggan</th><th className="p-3">Tanggal</th><th className="p-3">Item</th><th className="p-3 text-right">Total</th><th className="p-3 text-center">Status</th><th className="p-3 print:hidden">Aksi</th></tr></thead><tbody className="divide-y">

        {filteredTrx.map(t => (

          <tr key={t.id} className="hover:bg-gray-50"><td className="p-3"><p className="font-bold">{t.customer}</p><p className="text-[10px] text-gray-400">{t.id}</p></td><td className="p-3"><p>{new Date(t.date).toLocaleDateString('id-ID')}</p></td><td className="p-3 text-gray-500 text-xs">{t.items.map(i=>`${i.qty}${i.unit} ${i.name}`).join(', ')}</td><td className="p-3 text-right font-bold">{formatRupiah(t.total)}</td><td className="p-3 text-center"><span className={`px-2 py-0.5 rounded text-[10px] font-bold ${t.status==='Lunas'?'bg-green-100 text-green-700':'bg-orange-100 text-orange-700'}`}>{t.status}</span></td><td className="p-3 print:hidden flex gap-2">{t.status==='Belum Lunas' && <button onClick={()=>onLoadToKasir(t)} className="bg-blue-100 text-blue-700 px-2 py-1 rounded text-xs font-bold">Bayar</button>}<button onClick={()=>onDeleteTransaction(t.id)} className="text-red-400 hover:text-red-600"><Trash2 size={16}/></button></td></tr>

        ))}

      </tbody></table></div>

      <div className="text-right font-bold text-lg p-2">Total Lunas: {formatRupiah(totalOmset)}</div>

    </div>

  );

}
